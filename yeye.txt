     * 9. REGRAS DE NEGÓCIO DO RBAC
     * 
     * - Um usuário pertence a apenas um perfil
     * - Um perfil pode ter múltiplos usuários
     * - Cada perfil tem um nível de acesso (1 a 10)
     * - Menus de nível superior só aparecem para perfis com nível >= menu.nivelMinimoAcesso
     * - Permissões CRUD são controladas por perfil (podeVisualizar, podeCriar, podeEditar, podeEliminar)
     * - Usuário ADMIN (nível 10) tem acesso irrestrito a todos os menus
     * - Usuário CLIENTE só vê menus relacionados à sua própria conta
     * - A estrutura de menus é carregada apenas uma vez por sessão
     * - Alterações em PerfilMenu exigem religamento do usuário para efeito
     * - Todo acesso a tela verifica permissão do usuário na sessão
     * - O menu é gerado dinamicamente através de um MenuBuilder baseado no perfil

     * 12. EXEMPLO DE GERAÇÃO DE MENU DINÂMICO
     * 
     * public class MenuBuilderService {
     *     
     *     public JMenuBar construirMenuBar(SessaoUsuario sessao) {
     *         JMenuBar menuBar = new JMenuBar();
     *         
     *         // Buscar menus raiz (menuPaiId = null) para o perfil do usuário
     *         List<Menu> menusRaiz = menuDAO.findMenusByPerfil(sessao.getPerfilId());
     *         
     *         for (Menu menuRaiz : menusRaiz) {
     *             JMenu menu = new JMenu(menuRaiz.getNome());
     *             
     *             // Buscar submenus
     *             List<Menu> submenus = menuDAO.findSubmenus(menuRaiz.getId(), sessao.getPerfilId());
     *             
     *             for (Menu submenu : submenus) {
     *                 JMenuItem menuItem = new JMenuItem(submenu.getNome());
     *                 menuItem.setIcon(new ImageIcon(submenu.getIcone()));
     *                 
     *                 // Ao clicar, abre a view correspondente
     *                 menuItem.addActionListener(e -> {
     *                     abrirView(submenu.getCaminhoClasse());
     *                 });
     *                 
     *                 menu.add(menuItem);
     *             }
     *             
     *             if (submenus.size() > 0) {
     *                 menuBar.add(menu);
     *             }
     *         }
     *         
     *         return menuBar;
     *     }
     *     
     *     private void abrirView(String caminhoClasse) {
     *         try {
     *             Class<?> classe = Class.forName(caminhoClasse);
     *             JFrame frame = (JFrame) classe.getDeclaredConstructor().newInstance();
     *             frame.setVisible(true);
     *         } catch (Exception ex) {
     *             // Tratar exceção
     *         }
     *     }
     * }
